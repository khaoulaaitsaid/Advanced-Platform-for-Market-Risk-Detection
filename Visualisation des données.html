# Heatmap de Corrlation
corr_matrix = df[['initial_budget', 'market_budget', 'cotation_provisoire', 'cotation_definitive', 'cotation_adjudicataires']].corr()
plt.figure(figsize=(8, 6))
plt.imshow(corr_matrix, cmap='coolwarm', interpolation='nearest')
plt.title('Heatmap de Corrlation')
plt.colorbar()
tick_marks = np.arange(len(corr_matrix.columns))
plt.xticks(tick_marks, corr_matrix.columns, rotation=45)
plt.yticks(tick_marks, corr_matrix.columns)
plt.tight_layout()
plt.show()
