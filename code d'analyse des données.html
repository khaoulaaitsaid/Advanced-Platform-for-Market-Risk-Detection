#Détection des Risques Budgétaires
def detect_budget_risk(initial_budget, cotation_provisoire, cotation_definitive, cotation_adjudicataires, model, threshold=0.1):
    predicted_market_budget = model.predict([[initial_budget, cotation_provisoire, cotation_definitive, cotation_adjudicataires]])[0]
    risk = abs(predicted_market_budget - initial_budget) / initial_budget
    return predicted_market_budget, risk, risk > threshold

# Exemple d'utilisation de la fonction de détection des risques
initial_budget_example = 200000
cotation_provisoire_example = 210000

print(f"Predicted Market Budget: {predicted_market_budget}")
